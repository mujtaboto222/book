-- Run this once in your Supabase SQL editor:

CREATE TABLE IF NOT EXISTS public.sessions (
  id           uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id      uuid NOT NULL,
  email        text,
  session_key  text NOT NULL,
  ip           text,
  last_seen    timestamptz DEFAULT now(),
  UNIQUE(user_id, session_key)
);

ALTER TABLE public.sessions ENABLE ROW LEVEL SECURITY;

-- Allow logged-in users to upsert their own session
CREATE POLICY "Users manage own sessions" ON public.sessions
  FOR ALL USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);

-- Allow admins to read all sessions (handled via service key or anon read)
-- For simplicity, grant anon select (admin page already requires admin login):
CREATE POLICY "Admin read all" ON public.sessions
  FOR SELECT USING (true);
